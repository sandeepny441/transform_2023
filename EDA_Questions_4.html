# 9 -- Assess Data Distribution: For numerical fields like ACCT_NBR, 
#  analyze the distribution of values to spot any unusual patterns that might suggest inaccuracies.

# Check data type and handle accordingly
if df['ACCT_NBR'].dtype in ['int64', 'float64']:
    # Analyzing Numerical Distribution
    print("Descriptive Statistics for Account Numbers:")
    print(df['ACCT_NBR'].describe())

    # Plotting the distribution with histogram and density plot
    plt.figure(figsize=(10, 6))
    df['ACCT_NBR'].hist(bins=50, alpha=0.5, color='blue', edgecolor='black')
    df['ACCT_NBR'].plot(kind='kde', color='red')
    plt.title('Distribution of Account Numbers')
    plt.xlabel('Account Number')
    plt.ylabel('Frequency')
    plt.grid(True)
    plt.show()

    # Optional: Identifying outliers using IQR
    Q1 = df['ACCT_NBR'].quantile(0.25)
    Q3 = df['ACCT_NBR'].quantile(0.75)
    IQR = Q3 - Q1
    outlier_condition = ((df['ACCT_NBR'] < (Q1 - 1.5 * IQR)) | (df['ACCT_NBR'] > (Q3 + 1.5 * IQR)))
    outliers = df[outlier_condition]
    print(f"Number of outliers in Account Numbers: {outliers.shape[0]}")

else:
    # Handling Non-Numeric Data as Categorical
    print("Account Number field is not numeric. Analyzing as categorical data.")
    value_counts = df['ACCT_NBR'].value_counts()
    print(f"Unique values in Account Number field: {value_counts.size}")
    plt.figure(figsize=(10, 6))
    value_counts.head(20).plot(kind='bar')  # Display top 20 to avoid clutter
    plt.title('Top 20 Account Numbers')
    plt.xlabel('Account Number')
    plt.ylabel('Frequency')
    plt.xticks(rotation=45)
    plt.show()
